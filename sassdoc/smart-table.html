.smart-table--row

scss 
.smart-table--row {
  text-align: center;

  &:nth-child(even) {
    background: $theme-main--light;
  }

  &:not(.smart-table--head):nth-child(odd) {
    background: white;
  }

  @media screen and (max-width: $break-med) {
    &:not(.smart-table--head) {
      display: flex;
      align-items: center;
      padding: 10px 0;

      /* justify content for IE layout */
      justify-content: space-between;
    }

    &.record-is-expanded {
      align-items: stretch;
    }
  }
}

Requires

.record-is-expanded

scss 
.record-is-expanded {
  @media screen and (max-width: $break-med) {
    flex-direction: column;
    position: relative;

    .smart-table--cell {
      text-align: left;
      border-bottom: 1px solid $theme-black;
      margin: 5px 0;
      white-space: normal;
      overflow: visible;
      text-overflow: clip;
      display: block;
      flex: none;

      //for IE. Has no effect on Chrome or Firefox
      float: right;

      &:first-child {
        /* Make room for collapse button */
        padding: 30px 0 10px 0;
      }
    }

    td:nth-last-of-type(2) {
      /* Hide last border */
      border-bottom: none;
    }

    .smart-table--expand {
      position: absolute;
      top: 10px;
    }

    .bit-btn-expand {
      &:after {
        /* Toggle the expand button to a collapse button */
        content: "-";
      }
    }

    .smart-table--inlineHeading {
      @media screen and (max-width: $break-med) {
        display: inline;
      }
    }
  }
}
