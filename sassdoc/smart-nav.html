.smart-nav

scss 
.smart-nav {
  border-bottom: 1px solid $theme-secondary--deep;
  background-color: $theme-main;
  position: relative;

  @include apply-gradient(to left);

  @media screen and (max-width: $break-med) {
    @include apply-gradient(to top);
  }
}
